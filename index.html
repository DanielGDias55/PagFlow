// --- FUNÇÕES DE FILTRO E ORDENAÇÃO --- function setFiltroStatus(status) { filtroStatus = status; // Atualiza a aparência dos botões document.querySelectorAll('.filter-btn').forEach(btn => { btn.classList.remove('active'); }); document.getElementById(`btn-${status}`).classList.add('active'); renderizarTabela(); } function toggleOrdemAlfabetica() { ordemCrescente = !ordemCrescente; renderizarTabela(); } // --- AÇÕES EM MASSA (BULK ACTIONS) --- function toggleRowSelection(id) { if (selecionados.has(id)) { selecionados.delete(id); } else { selecionados.add(id); } renderizarTabela(); } function toggleAll(source) { if (source.checked) { dadosLocais.forEach(p => selecionados.add(p.id)); } else { selecionados.clear(); } renderizarTabela(); } function clearSelection() { selecionados.clear(); const selectAllCheck = document.getElementById('selectAll'); if (selectAllCheck) selectAllCheck.checked = false; renderizarTabela(); } function updateBulkUI() { const bar = document.getElementById('bulkActionsBar'); if (!bar) return; if (selecionados.size > 0) { document.getElementById('bulkCount').innerText = selecionados.size; // Mostra a barra bar.classList.remove('translate-y-20', 'opacity-0', 'pointer-events-none'); bar.classList.add('translate-y-0', 'opacity-100'); } else { // Esconde a barra e limpa classes de visibilidade bar.classList.add('translate-y-20', 'opacity-0', 'pointer-events-none'); bar.classList.remove('translate-y-0', 'opacity-100'); } } function excluirSelecionados() { if (confirm(`Tem certeza que deseja excluir ${selecionados.size} registros?`)) { const promises = []; selecionados.forEach(id => { promises.push(db.ref(`cobrancas/${mesAtual}/${id}`).remove()); }); Promise.all(promises).then(() => { selecionados.clear(); alert("Registros excluídos com sucesso!"); renderizarTabela(); }); } } // --- EXPORTAÇÃO --- function exportarExcel() { if (dadosLocais.length === 0) return alert("Não há dados para exportar."); let csv = "ID,Nome,Telefone,Valor,Adiantadas,Status\n"; dadosLocais.forEach(p => { csv += `${p.id},${p.nome},${p.tel || ''},${p.valor},${p.adiantadas || 0},${p.pago ? 'Pago' : 'Pendente'}\n`; }); const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' }); const link = document.createElement("a"); link.href = URL.createObjectURL(blob); link.setAttribute("download", `pool_manager_${mesAtual}.csv`); link.click(); }
